#this prints the value 100 feet too low, it takes the 5600 foot value and I want the 5700 foot value
def getTakeoffDistance(densityAltitude):
    # Find the thousands of feet and the hundreds of feet
    thousands = densityAltitude // 1000 * 1000
    hundreds = (densityAltitude % 1000) // 100 * 100
    

    if thousands in baseTODist and hundreds in baseTODist[thousands]:
        return baseTODist[thousands][hundreds]

    return None
